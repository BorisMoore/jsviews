HTML:

<td>
	{{#sort Languages reverse="true"}}
		<div>{{=Name}}</div>
	{{/sort}}
</td>

Script:

$.views.registerTags({

	sort: function( array ){
		...
		if ( this.reverse === "true" ) {
			// Render in reverse order
			for ( var l = array.length, i = l; i; i-- ) {
				ret += $.render( this.tmpl, array[ i - 1 ]);
			}
		}
		...
		return ret;
	}

});
